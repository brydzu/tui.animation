    var container = $('.container');
var box = $('.box');
var basePos = container.position();
var runner = null;
var easing = 'easeInOutQuad';

function getPos(x, y) {
    return [x - basePos.left, y - basePos.top];
}

$('select').on('change', function(e) {
    easing = $(e.target).find(':selected').val();
});

container.on('mousedown', function(e) {
    var to = getPos(e.clientX, e.clientY);

    if (runner) {
        runner.cancel();
    }

    var from = box.position();

    runner = tui.component.animation.anim({
        from: [from.left, from.top],
        to: to,
        duration: 300,
        easing: easing,
        frame: function(x, y) {
            box.css({
                left: x + 'px',
                top: y + 'px'
            });
        }
    });

    runner.run();
});
