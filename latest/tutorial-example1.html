    • JS

    $('div.context').each(function(_, container) {
    container = $(container);

    var easing = container.attr('id');
    var box = container.find('.move-box');

    container.find('.move-btn').on('click', function() {
        tui.component.animation.anim({
            easing: easing,
            frame: function(left) {
                box.css({left: left + '%'});
            }
        }).run();
    });
});
